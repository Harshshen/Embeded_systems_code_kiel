ORG 0000H      
MOV A, #38H                  //SELECTING LCD DISPLAY
ACALL CSR    
MOV A, #0EH                  //STARTING BLINKING              
ACALL CSR     
MOV A, #85H                 //FORCE CURSOR TO START
ACALL CSR  
MOV A, #01H                 //CLEARING SCREEN
ACALL CSR   

MOV A, #85H                //TO GET COURSER IN THE MIDDLE
ACALL CSR         
MOV DPTR ,#LOne      //MOVING STRING  1 LONE
REV : MOV A,#00H   
MOVC A,@A+DPTR    
JZ NXT                          //CHECKING ACCUMULATOR DATA IS ZERO OR NOT
ACALL DSR                 //CALLING DATA SERVICE ROUTINE
INC DPTR    
SJMP REV 
   
NXT:MOV A,#0C4H      //GETTING CURSOR DOWN SECOND LINE
ACALL CSR                  // CALLING COMMAND SERVICE ROUTINE
MOV DPTR ,#LTwo       // SETTING STRING 2 FOR OUTPUT AFTER STRING 1
REPEAT : MOV A,#00H   
MOVC A,@A+DPTR    
JZ FINISH      
ACALL DSR                 //CALLING DATA  SERVICE ROUTINE
INC DPTR                    //INCREASE DPTR 
SJMP REPEAT            //JUMP BACK TO REPEAT
FINISH: SJMP FINISH 


CSR: ACALL DEL_ROUTINE   // COMMAND SERVICE ROUTINE
MOV P1, A   
CLR P2.1   
SETB P2.2
CLR P2.2
RET                                        //RETURNING TO MAIN PROGRAM

DSR: ACALL DEL_ROUTINE      //DATA SERVICE ROUTINE
MOV P1, A      
SETB P2.1     
SETB P2.2     
CLR P2.2     
RET       

DEL_ROUTINE: MOV R0, #0FFH    //DELAY ROUTINE
L1: MOV R1, #0FFH   
L2: DJNZ R1, L2    
DJNZ R0, L1     
RET   


LOne : DB 'IIITN',0                        //STRING 1
LTwo : DB 'ECE DEPT.',0	   //STRING 2
END
